import { useRouter } from "next/router";
import books from "../../data/semester2.json";
export default () => {
  const router = useRouter();
  const post = books[router.query.book];
  if (!post) return <p></p>;

  return (
    <div className="index">
      <h1>{post.name}</h1>
      {Object.entries(books).map((value, index) => {
        return (
        <li key={index}>{value[1].name}</li>
        );
      })}
      <li>hello</li>
    </div>
  );
};
